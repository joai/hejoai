		if(pageNumber != parseInt(pageNumber)){
			$.addtip({type: "error", message: "跳转页码必须为数值", autoclose: 3});
			$(obj).prev().prev().focus();
			return false;

		if(pageNumber &lt; 1){
			pageNumber = 1;
		}
		var url = "/admin/business/log?searchType=1&amp;search=&amp;startAt=&amp;endAt=&amp;page={page}";
		url = url.replace("\{page\}", pageNumber);
		window.location = url;
	}
